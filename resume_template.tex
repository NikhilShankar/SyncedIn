% ---------- CUSTOM SECTION TEMPLATES ----------

% Custom Section Template 1: Simple (like Professional Summary)
% Parameters: #1=Title, #2=Subtitle (optional, left-aligned below title), #3=Subtitle_right (optional, right-aligned), #4=Content
\newcommand{\customSectionTemplateOne}[4]{
  \section{#1}
  \ifx&#2&% Check if subtitle is empty
    % No subtitle
  \else
    % Has subtitle (left-aligned below title)
    \textit{\normalfont{#2}} \\
    \vspace{4pt}
  \fi
  \normalfont{#4}
}

% Custom Section Template 2: With Subsections (no bullets)
% Parameters: #1=Section Title
% Content added dynamically via items
\newcommand{\customSectionTemplateTwo}[1]{
  \section{#1}
  \begin{itemize}[leftmargin=0in, label={}]
}
% Helper for subsection items (title, subtitle, subtitle_right, content)
% #1=Title (bold, left), #2=Subtitle (italic, below title, left), #3=Subtitle_right (normal, right-aligned), #4=Content
\newcommand{\customSubsectionItemTwo}[4]{
  \item
    \begin{tabular*}{\textwidth}[t]{@{}l@{\extracolsep{\fill}}r@{}}

      \textbf{#1} & \normalfont #3 \\
    \end{tabular*}
    \ifx&#2&% Check if subtitle is empty
      % No subtitle
    \else
      \vspace{2pt}
      \noindent\textit{\normalfont{#2}} \\
    \fi
    \vspace{2pt}
    \noindent\normalfont{#4}
    \vspace{6pt}
}
\newcommand{\customSectionTemplateTwoEnd}{
  \end{itemize}
}
